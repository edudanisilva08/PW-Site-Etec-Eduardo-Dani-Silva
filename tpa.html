Projeto de arrays realizado em TPA, como auxilio da professora Elisângela
JS

<h1>Trabalho em Dupla: Eduardo e Pacifico</h1> <div class="container"> <div id="array1" class="box"></div> <div id="array2" class="box"></div> </div> <div class="botaos"> <div class="botoes1"> <button onclick="removeArray1()">Remover de Eduardo</button> <button onclick="removeArray2()">Remover de Pacifico</button> </div> <div class="botoes2"> <button onclick="addArray1()">Adicionar a Eduardo</button> <button onclick="addArray2()">Adicionar a Pacifico</button> </div> </div> <script> let array1 = [2, 3, 1]; let array2 = [3, 2, 1]; function mostraArrays() { document.getElementById("array1").innerText = "Eduardo:\n" + array1.join("\n"); document.getElementById("array2").innerText = "Pacifico:\n" + array2.join("\n"); } function addArray1() { let newElement = parseInt(prompt("Digite um novo número para o Eduardo:")); if (newElement) { array1.push(newElement); array1.sort((a, b) => b - a); mostraArrays(); } } function removeArray1() { let removeType = prompt("Digite "+"pop"+" para remover do final ou "+"shift"+" para remover do início:"); if (removeType === "pop") { array1.pop(); array1.sort((a, b) => b - a); } else if (removeType === "shift") { array1.shift(); } mostraArrays(); } function addArray2() { let newElement = parseInt(prompt("Digite um novo número para o Pacifico:")); if (newElement) { array2.push(newElement); array2.sort((a, b) => b - a); mostraArrays(); } } function removeArray2() { let removeType = prompt("Digite "+"pop"+" para remover do final ou "+"shift"+" para remover do início:"); if (removeType === "pop") { array2.pop(); array2.sort((a, b) => b - a); } else if (removeType === "shift") { array2.shift(); array2.sort((a, b) => b - a); } mostraArrays(); } mostraArrays(); </script>
